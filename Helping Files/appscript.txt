/*************************************************
 * SCC CERTIFICATE PORTAL API
 * Source tab: Verified Students
 * Form tab: Form responses 1
 *************************************************/

const VERIFIED_STUDENTS_SHEET_NAME = "Verified Students";

function doGet(e) {
  try {
    const action = String((e && e.parameter && e.parameter.action) || "list").toLowerCase();

    if (action === "list" || action === "" || action === "get" || action === "read") {
      return output({ students: getVerifiedStudents_() });
    }

    if (action === "generate") {
      return output(generateCertificate_(Number(e.parameter.rowNumber)));
    }

    return output({ students: getVerifiedStudents_() });
  } catch (err) {
    return output({ error: err.message || String(err) });
  }
}

function getVerifiedStudents_() {
  const sheet = getVerifiedStudentsSheet_();
  const values = sheet.getDataRange().getValues();
  if (values.length < 2) return [];

  const headerMap = buildHeaderMap_(values[0]);
  const students = [];

  for (let r = 1; r < values.length; r++) {
    const row = values[r];
    if (row.every((cell) => String(cell || "").trim() === "")) continue;
    students.push(mapStudent_(row, headerMap, r + 1));
  }

  return students;
}

function generateCertificate_(rowNumber) {
  if (!rowNumber || rowNumber < 2) {
    throw new Error("Valid rowNumber is required.");
  }

  const sheet = getVerifiedStudentsSheet_();
  const values = sheet.getDataRange().getValues();
  const headerMap = buildHeaderMap_(values[0]);
  const row = values[rowNumber - 1];

  if (!row) throw new Error("Student row not found.");

  const certCol = ensureColumn_(sheet, headerMap, "Certificate ID");
  const generatedCol = ensureColumn_(sheet, headerMap, "Generated");
  const existingCertId = getCellAny_(row, headerMap, ["certificate id"]);
  const certId = existingCertId || createCertificateId_(rowNumber);

  sheet.getRange(rowNumber, certCol + 1).setValue(certId);
  sheet.getRange(rowNumber, generatedCol + 1).setValue("Yes");

  const updatedHeaders = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0];
  const updatedMap = buildHeaderMap_(updatedHeaders);
  const updatedRow = sheet.getRange(rowNumber, 1, 1, sheet.getLastColumn()).getValues()[0];

  return {
    success: true,
    student: mapStudent_(updatedRow, updatedMap, rowNumber)
  };
}

function getVerifiedStudentsSheet_() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(VERIFIED_STUDENTS_SHEET_NAME);

  if (!sheet) {
    throw new Error('Sheet tab "Verified Students" not found. Please keep this exact tab name.');
  }

  return sheet;
}

function mapStudent_(row, headerMap, rowNumber) {
  return {
    rowNumber: rowNumber,
    name: getCellAny_(row, headerMap, ["name", "full name"]),
    email: getCellAny_(row, headerMap, ["email", "email address"]),
    mobile: getCellAny_(row, headerMap, ["mobile", "mobile number"]),
    course: getCellAny_(row, headerMap, ["course", "course name"]),
    joiningDate: formatForJson_(getCellAny_(row, headerMap, ["joining date", "course joining date"])),
    endingDate: formatForJson_(getCellAny_(row, headerMap, ["ending date", "course completion date"])),
    courseReview: getCellAny_(row, headerMap, ["course review"]),
    verificationStatus: getCellAny_(row, headerMap, ["verification status"]),
    certificateId: getCellAny_(row, headerMap, ["certificate id"]),
    generated: getCellAny_(row, headerMap, ["generated"]),
    remarks: getCellAny_(row, headerMap, ["remarks", "any feedback"])
  };
}

function buildHeaderMap_(headers) {
  const map = {};
  headers.forEach((header, index) => {
    map[normalizeHeader_(header)] = index;
  });
  return map;
}

function getCellAny_(row, headerMap, headerNames) {
  for (let i = 0; i < headerNames.length; i++) {
    const index = headerMap[normalizeHeader_(headerNames[i])];
    if (index !== undefined) {
      const value = row[index];
      return value === null || value === undefined ? "" : value;
    }
  }
  return "";
}

function ensureColumn_(sheet, headerMap, headerName) {
  const normalized = normalizeHeader_(headerName);
  if (headerMap[normalized] !== undefined) return headerMap[normalized];

  const nextCol = sheet.getLastColumn() + 1;
  sheet.getRange(1, nextCol).setValue(headerName);
  headerMap[normalized] = nextCol - 1;
  return nextCol - 1;
}

function createCertificateId_(rowNumber) {
  const year = new Date().getFullYear();
  return "SAMYAK-" + year + "-" + String(rowNumber - 1).padStart(4, "0");
}

function formatForJson_(value) {
  if (Object.prototype.toString.call(value) === "[object Date]" && !isNaN(value)) {
    return Utilities.formatDate(value, Session.getScriptTimeZone(), "yyyy-MM-dd");
  }
  return value;
}

function normalizeHeader_(value) {
  return String(value || "").trim().toLowerCase().replace(/\s+/g, " ");
}

function output(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
